FFC6 7D                 LD      A,L             ; Get column

FFCD E5                 PUSH    HL              ; Re-save column
FFCE C5                 PUSH    BC              ; Put back return
FFCF 7B                 LD      A,E             ; Get row
FFD0 C9                 RET

FFD4 7E                 LD      A,(HL)          ; Get byte
FFD5 C9                 RET

FFDE C310E9             JP      GETHDR          ; CLOAD - Get header
